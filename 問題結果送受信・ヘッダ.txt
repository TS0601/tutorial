var getPageStr = function(url) {
    // URL文字列を「/」区切りで分解
    var arg = url.split("/");
    // 末尾の要素を取り出し
    var ret = arg.slice(-1)[0];
    // 末尾が空の場合は更に一つ前の要素を取り出し
    if(ret === "" && arg.length > 1) {
        ret = arg.slice(-2)[0];
    }
    // ?, #, &より前の要素のみを取り出し
    ret = ret.split("?")[0];
    ret = ret.split("#")[0];
    ret = ret.split("&")[0];
	return ret;
};
